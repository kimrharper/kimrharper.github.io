Playing with wave files¶

Librosa loads and decodes the audio as a time series y, represented as a one-dimensional NumPy floating point array. The variable sr contains the sampling rate of y, that is, the number of samples per second of audio. By default, all audio is mixed to mono and resampled to 22050 Hz at load time. This behavior can be overridden by supplying additional arguments to librosa.load().

def mfcc_extract(filename):
    n_mfcc     = 40
    hop_length = 12800
    
    # y -> time series; sr -> sample rate per second
    fn    = filename +".wav"
    y, sr = librosa.load(fn)   
    
    # Compute MFCC features from the raw signal (https://en.wikipedia.org/wiki/Mel-frequency_cepstrum)
    # In sound processing, the mel-frequency cepstrum (MFC) is a representation of the short-term power spectrum of a sound,
    # based on a linear cosine transform of a log power spectrum on a nonlinear mel scale of frequency.
    mfcc = librosa.feature.mfcc(y=y, sr=sr, hop_length=hop_length, n_mfcc=n_mfcc)
    
    # Normalize mfcc for eliminating bias for lower frequencies
    # mfcc = librosa.util.normalize(mfcc, norm=inf, axis=0, threshold=None, fill=None)
    
    df_ = pd.DataFrame(np.transpose(mfcc), columns = ["n_mfcc_{:02d}".format(i) for i in range((mfcc.shape)[0])])
    
    return df_

##Statisitc metric function

def statsex(df_):
    
    df_col = df_.columns
    
    stat_names = ("stat_mean","stat_mode","stat_median","stat_tmean","stat_q1","stat_q3"
        ,"stat_std","stat_var","stat_iqr","stat_min","stat_max","stat_range"
        ,"stat_skew","stat_kurtosis")
    
    stat_funcs = ("np.mean(df_.iloc[:,{}])"
                  ,"stats.mode(df_.iloc[:,{}])[0][0]"
                  ,"np.median(df_.iloc[:,{}])"
                  ,"stats.trim_mean(df_.iloc[:,{}],0.05)"
                  ,"np.percentile(df_.iloc[:,{}],25)"
                  ,"np.percentile(df_.iloc[:,{}],75)"
                  ,"np.std(df_.iloc[:,{}], ddof=1)"
                  ,"np.var(df_.iloc[:,{}], ddof=1)"
                  ,"stats.iqr(df_.iloc[:,{}])"
                  ,"np.min(df_.iloc[:,{}])"
                  ,"np.max(df_.iloc[:,{}])"
                  ,"np.ptp(df_.iloc[:,{}])"
                  ,"stats.skew(df_.iloc[:,{}])"
                  ,"stats.kurtosis(df_.iloc[:,{}])")

    
    #Creating a dictionary for the purpose of aggregating features across the attributes mentioned in cols with stat_funcs
    feature_space  = {}
    
    for col in range(len(df_col)):
        for name,func in zip (stat_names,stat_funcs):
            key = df_col[col] + "_" + name
            feature_space[key] = eval(func.format(col)) 
    
    return feature_space

def change_amplitude(filename):  

#   y -> time series; sr -> sample rate per second
    fn    = filename+'.wav'
    y, sr = librosa.load(fn)
    delta_y = [abs(y[i]-y[i-1]) for i in range(1,len(y))]
    return np.mean(delta_y)

## Minkowskimatrices
def minkowski_mat(df,p):
    min_mat = []
    for idx,col in enumerate(df.columns):
        mean = np.full((df.shape[0],1),df.mean()[idx])
        vals = df.iloc[:,idx]
        dist = scipy.spatial.distance.minkowski(mean, vals, p=p, w=None)
        min_mat.append([col, dist])
    
    return min_mat
